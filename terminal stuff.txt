## step 1: set up 
# check wd is correct 
cd path/to/shark-ray-scraper

touch validator_app.py # creates a blank python file in current folder 

# open VS code

code validator_app.py # opens the file 

# work on the file in VS code 

# back in terminal 
# run it 
streamlit run validator_app.py
